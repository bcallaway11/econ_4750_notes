3.15 for loops

  • compute sum of series 1 to n
compute_s_n <- function(n){
  x <- 1:n
  sum(x)
}

m <- 25
s_n <- vector(length = m) # create an empty vector
for(n in 1:m){
  s_n[n] <- compute_s_n(n)
}

n <- 1:m
plot(n, s_n)

3.16 Vectorization

  • often preferred over for loops
n <- 1:25
compute_s_n(n)
## Warning in 1:n: numerical expression has 25 elements: only the first used
## [1] 1
n <- 1:25
s_n <- sapply(n, compute_s_n)
